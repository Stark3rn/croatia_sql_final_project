CREATE TABLE customer (
id BIGSERIAL PRIMARY KEY,
firstName VARCHAR(25),
lastName VARCHAR(25),
email VARCHAR(50),
address VARCHAR(100)
);

CREATE TABLE product (
id BIGSERIAL PRIMARY KEY,
name VARCHAR(50),
description VARCHAR(100),
price MONEY,
stockQuantity BIGINT
);

CREATE TABLE "order" (
id BIGSERIAL PRIMARY KEY,
clientID BIGINT REFERENCES Customer(id),
date DATE,
status VARCHAR(15),
amount MONEY
);

CREATE TABLE orderitem (
id BIGSERIAL PRIMARY KEY,
orderID BIGINT REFERENCES "Order"(id),
productID BIGINT REFERENCES Product(id),
quantity INT,
unitPrice MONEY
);

CREATE TABLE payment (
id BIGSERIAL PRIMARY KEY,
orderID BIGINT REFERENCES "Order"(id),
amount MONEY,
date DATE,
paymentPlatform VARCHAR(50)
);
